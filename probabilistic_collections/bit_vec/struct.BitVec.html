[−][src]Struct probabilistic_collections::bit_vec::BitVec

pub struct BitVec { /* fields omitted */ }

A growable list of bits implemented using a Vec<u8>

Examples

use probabilistic_collections::bit_vec::BitVec;

let mut bv = BitVec::from_elem(5, false);

bv.set(0, true);
bv.set(1, true);
bv.set(2, true);
assert_eq!(
    bv.iter().collect::<Vec<bool>>(),
    vec![true, true, true, false, false],
);

bv.set_all(true);
assert_eq!(
    bv.iter().collect::<Vec<bool>>(),
    vec![true, true, true, true, true],
);

bv.flip(0);
bv.flip_all();
assert_eq!(
    bv.iter().collect::<Vec<bool>>(),
    vec![true, false, false, false, false],
);

bv.push(true);
assert_eq!(
    bv.iter().collect::<Vec<bool>>(),
    vec![true, false, false, false, false, true],
);
assert_eq!(bv.pop(), Some(true));

let clone = bv.clone();
bv.flip_all();
bv.union(&clone);
assert_eq!(
    bv.iter().collect::<Vec<bool>>(),
    vec![true, true, true, true, true],
);
